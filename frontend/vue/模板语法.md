# 模板语法
```vue
<template>
<!-- 1. 文本插值 -->
<span>Message: {{ msg }}</span>

<!-- 2. 原始HTML: v-html （不建议使用）-->
<p>Using text interpolation: {{ rawHtml }}</p>
<p>Using v-html directive: <span v-html="rawHtml"></span></p>

<!-- 3. 属性绑定: v-bind -->
<div v-bind:id="house">house</div>
<div :id="tree">tree</div>
<div :id>tree</div>
<!-- 布尔型属性 -->
<button :disabled="isButtonDisabled">Button</button>
<!-- 动态绑定多个属性 -->
<div v-bind="objectOfAttrs"></div>

<!-- 如何绑定class？ -->
<!-- 绑定对象 -->
<div 
    class="static"
    :class="{ active: isActive }">
</div>
<div :class="classObject"></div> 
<!-- 绑定数组 -->
<div :class="[activeClass, errorClass]"></div>

<!-- 如何绑定style？ -->
<div :style="{ color: activeColor, fontSize: fontSize + 'px' }"></div>
<div :style="styleObject"></div>
<!-- 同时绑定多个样式对象 -->
<div :style="[baseStyles, overridingStyles]"></div>


<!-- 4. 条件渲染: v-if、v-show -->
<button @click="awesome = !awesome">Toggle</button>
<h1 v-if="awesome">Vue is awesome!</h1>
<h1 v-else>Oh no 😢</h1>
<!-- v-if 和 v-show的区别？v-show修改display属性，v-if是按条件渲染 -->
<h1 v-show="awesome">Hello!</h1>


<!-- 5. 列表渲染: v-for -->

<!-- 6. 事件处理: v-on -->
<!-- 内联事件处理器 -->
<button @click="count++">Add 1</button>
<p>Count is: {{ count }}</p>
<!-- `greet` 是上面定义过的方法名 -->
<button @click="greet">Greet</button>

<!-- 6. ref 标签 -->
<!-- 如果使用id，不同组件可能id会冲突 -->
<h2 ref="title"></h2>


</template>

<script setup>
// {{ }} 中写的也是js表达式
const msg = "hello"
const awesome = ref(true)
const count = ref(0)

// 如果将ref加在组件上，那么就能拿到实例对象，但是能看到什么需要儿子允许
// 儿子怎么允许呢？defineExpose
const title = ref()
console.log(title.value) // 打印DOM元素


// v- 指令后面是js表达式
// <div id="ok"></div>      ok 是字符串
// <div :id="ok"></div>     ok 是变量
const rawHtml = '<span style="color: red">This should be red.</span>'

// 如何绑定一个属性的值？
// 普通的属性比如id，绑定的值都是字符串
const house = "house"
const tree = "tree"
const id = "xxx"
const isButtonDisabled = true
const objectOfAttrs = {
  id: 'container',
  class: 'wrapper',
  style: 'background-color:green'
}

// class和style比较特殊，除了可以绑定字符串，还可以绑定对象、数组
// 绑定class
const isActive = ref(true)

const classObject = reactive({
  active: true,
  'text-danger': false
})

const activeClass = ref('active')
const errorClass = ref('text-danger')

// 绑定style
const activeColor = ref('red')
const fontSize = ref(30)

const styleObject = reactive({  // 直接绑定样式对象
  color: 'red',
  fontSize: '30px'
})

// 方法事件
const name = ref('Vue.js')

function greet(event) {
  alert(`Hello ${name.value}!`)
  // `event` 是 DOM 原生事件
  if (event) {
    alert(event.target.tagName)
  }
}
<script>
```
